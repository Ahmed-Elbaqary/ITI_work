1) Shell basic commands:
-----------------------
[root@master ~]#pwd                (print working directory)
[root@master ~]#date               (show current date 24Hr Display)
[root@master ~]#cal                (show Calender of this day,month,year)
[root@master ~]#cal 2020           (show Calender of whole year 2020)
[root@master ~]#cal 3 2020         (show Calender of month 3 year 2020)
[root@master Desktop]#cd           (change director to home directory of the current user)
[root@master ~]#whoami             (print current login user)

*Note*: ~ is a variable its value home directory of the login user

[root@master ~]#ls                 (list content of the current working directry "files, directories")
[root@master ~]#ls -l              (list content of the current working directry as long list option "more details")

*Note*: we put - before any option in most of commands

[root@master ~]#ls -a              (list all content of the current working directry "hidden files")
[root@master ~]#ls --all           (list all content of the current working directry "hidden files") *SAME*
[root@master ~]#ls -la             (list all content of the current working directry "as long list option with more details" and "hidden files")
[root@master ~]#ls -la /home       (list all content of the specific directry "as long list option with more details" and "hidden files")

*Note*: command -[option] [argument]

[root@master ~]#clear              (or you can use "Ctrl+l" to clear screen of terminal)
[root@master ~]#reset              (clear screen and reset terminal)
[root@master ~]#exit               (or you can use "Ctrl+d" to close terminal)

[root@master ~]#ls --help          (*Note*: to know all options about specific command use option --help)
[root@master ~]#date --help

Usage: date [OPTION]... [+FORMAT]
  or:  date [-u|--utc|--universal] [MMDDhhmm[[CC]YY][.ss]]
[ ] surround optional items
... represents an arbitrary-length list of items of that type
| means only one of them can be specified
<> represents variable data. For example <filename>

4) Dsiplay file content commands:
--------------------------------
[root@master ~]#cat /etc/passwd                (this command to show the content inside the file)  Later
[root@master Desktop]# head /etc/passwd        (show first 10 lines by defualt of file)
[root@master Desktop]# head -n 3 /etc/passwd   (show first 3 lines of file)
[root@master Desktop]# head -3 /etc/passwd     (same as above)
[root@master Desktop]# head /etc/passwd -n 3   (same as above)
[root@master Desktop]# tail /etc/passwd        (show last 10 lines by defualt of file)
[root@master Desktop]# tail -n 3 /etc/passwd   (show last 3 lines of file)
[root@master Desktop]# tail -3 /etc/passwd     (same as above)
[root@master Desktop]# tail /etc/passwd -n 3   (same as above)
[root@master Desktop]# wc /etc/passwd          (word count--> used to display number LINES of WORDS and CHARACTERS in file)

[root@master Desktop]# wc -l /etc/passwd       (display only number of lines )
[root@master Desktop]# wc -w /etc/passwd       (display only number of words )
[root@master Desktop]# wc -c /etc/passwd       (display only number of characters )

[root@master Desktop]# wc -lwc /etc/passwd

6) Command history:
--------------------------------
[root@master ~]# cat .bash_history       (hidden file to store history of user commands)
[root@master ~]# history                 (command to show history of user commands)
[root@master ~]# !88                     (command to execute specific command with 88 number in history)
[root@master ~]# !-10                    (from tail)
[root@master ~]# !ls                     (the most recent command)
[root@master ~]# !!
[root@master ~]# history -c              (to clear the history)
 
*Note*: new command are loaded into buffer and shown in the history command when we exit the shell
